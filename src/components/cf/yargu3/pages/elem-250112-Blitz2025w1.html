Задача A. Башни

(Время 2с. Память 256мб)
Решение 
#include <iostream>

using namespace std;

int main() {
  int n; // число брусков
  cin >> n;
  int A[1001] = {};
  int in;

  for (int i = 1; i <= n; i++) {
    cin >> in; // длина бруска
    A[in]++;
  }

  int maxa = 1; // высота наибольшей башни
  int cnt = 0; // число башен
  for (int i = 1; i < 1001; i++) {
    if (A[i]) {
      cnt++;
    }
    if (A[i] > maxa) {
      maxa = A[i];
    }
  }

  cout << maxa << ' ' << cnt;
}

Задача B. Малинки

(Время 2с. Память 256мб)
Решение

Посчитаем решение для каждого k отдельно. При k=4 нужно аккуратно обработать варианты.

#include <iostream>

using namespace std;

int main() {
  // условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  int t; // число тестов
  cin >> t;

  while (t--) {
    int n; // длина массива
    int k; // от 2 до 5
    cin >> n >> k;
    int in;

    if (k == 2) {
      int flag2 = 1;

      for (int i = 0; i < n; i++) {
        cin >> in;
        if (flag2 && in % 2 == 0) {
          flag2 = 0;
        }
      }

      if (flag2) {
        cout << 1 << '\n';
      } else {
        cout << 0 << '\n';
      }

    } else if (k == 3) {
      int flag0 = 0;
      int flag1 = 0;
      int flag2 = 0;

      for (int i = 0; i < n; i++) {
        cin >> in;
        if (in % 3 == 0) {
          flag0 = 1;
        } else if (in % 3 == 2) {
          flag1 = 1;
        } else if (in % 3 == 1) {
          flag2 = 1;
        }
      }

      if (flag0) {
        cout << 0 << '\n';
      } else if (flag1) {
        cout << 1 << '\n';
      } else if (flag2) {
        cout << 2 << '\n';
      }

    } else if (k == 4) {
      int flag0 = 0;
      int flag1 = 0;
      int chet = 0;


      for (int i = 0; i < n; i++) {
        cin >> in;
        if (in % 4 == 0) {
          flag0 = 1;
        } else if (in % 4 == 3) {
          flag1 = 1;
        } else if (in % 2 == 0) {
          if (chet) {
            flag0 = 1;
          }
          chet = 1;
          flag1 = 1;
        }
      }

      if (flag0) {
        cout << 0 << '\n';
      } else if (flag1) {
        cout << 1 << '\n';
      } else {
        cout << 2 << '\n';
      }

    } else if (k == 5) {
      int flag0 = 0;
      int flag1 = 0;
      int flag2 = 0;
      int flag3 = 0;
      int flag4 = 0;

      for (int i = 0; i < n; i++) {
        cin >> in;
        if (in % 5 == 0) {
          flag0 = 1;
        } else if (in % 5 == 4) {
          flag1 = 1;
        } else if (in % 5 == 3) {
          flag2 = 1;
        } else if (in % 5 == 2) {
          flag3 = 1;
        } else if (in % 5 == 1) {
          flag4 = 1;
        }
      }

      if (flag0) {
        cout << 0 << '\n';
      } else if (flag1) {
        cout << 1 << '\n';
      } else if (flag2) {
        cout << 2 << '\n';
      } else if (flag3) {
        cout << 3 << '\n';
      } else if (flag4) {
        cout << 4 << '\n';
      }
    }
  }
}
