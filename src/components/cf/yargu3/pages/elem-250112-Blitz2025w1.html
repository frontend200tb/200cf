Задача A. Башни

(Время 2с. Память 256мб)
Решение 
#include <iostream>

using namespace std;

int main() {
  int n; // число брусков
  cin >> n;
  int A[1001] = {};
  int in;

  for (int i = 1; i <= n; i++) {
    cin >> in; // длина бруска
    A[in]++;
  }

  int maxa = 1; // высота наибольшей башни
  int cnt = 0; // число башен
  for (int i = 1; i < 1001; i++) {
    if (A[i]) {
      cnt++;
    }
    if (A[i] > maxa) {
      maxa = A[i];
    }
  }

  cout << maxa << ' ' << cnt;
}
