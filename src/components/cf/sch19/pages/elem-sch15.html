Задача A. HQ9+

Решение 
#include <iostream>
#include <string>

using namespace std;

int main() {
  string s; // входная строка
  getline(cin, s);
  bool flag = false;

  for (char ch : s) {
    if (ch == 'H' || ch == 'Q' || ch == '9') {
      flag = true;
    }
  }

  if (flag) {
    cout << "YES";
  } else {
    cout << "NO";
  }
}

Задача B. Счастливый билет

Решение 
#include <iostream>

using namespace std;

int main() {
  int n; // длина номера
  cin >> n;

  char in;
  bool flag = true;
  int sum1 = 0;
  for (int i = 0; i < n / 2; i++) {
    cin >> in;
    if (in != '4' && in != '7') {
      flag = false;
    }
    int tmp = static_cast<int>(in);
    sum1 += tmp;
  }

  int sum2 = 0;
  for (int i = 0; i < n / 2; i++) {
    cin >> in;
    if (in != '4' && in != '7') {
      flag = false;
    }
    int tmp = static_cast<int>(in);
    sum2 += tmp;
  }

  if (sum1 != sum2) {
    flag = false;
  }

  if (flag) {
    cout << "YES";
  } else {
    cout << "NO";
  }
}
