Задача A. HQ9+

Решение 
#include <iostream>
#include <string>

using namespace std;

int main() {
  string s; // входная строка
  getline(cin, s);
  bool flag = false;

  for (char ch : s) {
    if (ch == 'H' || ch == 'Q' || ch == '9') {
      flag = true;
    }
  }

  if (flag) {
    cout << "YES";
  } else {
    cout << "NO";
  }
}

Задача B. Счастливый билет

Решение 
#include <iostream>

using namespace std;

int main() {
  int n; // длина номера билета от 2 до 50
  cin >> n;

  string s; // номер билета от 2 до 50 знаков
  cin >> s;

  bool flag = true; // номер счастливый

  // решение и вывод результата
  int n2 = n / 2; // половина цифр
  int sum1 = 0; // сумма цифр первой половины
  int sum2 = 0; // сумма цифр второй половины

  for (int i = 0; i < n2; i++) {
    if (s[i] == '4' || s[i] == '7') {
      sum1 += s[i] - '0';
    } else {
      flag = false;
    }
  }

  for (int i = n2; i < n; i++) {
    if (s[i] == '4' || s[i] == '7') {
      sum2 += s[i] - '0';
    } else {
      flag = false;
    }
  }

  if (sum1 != sum2) {
    flag = false;
  }

  if (flag) {
    cout << "YES";
  } else {
    cout << "NO";
  }
}

Задача D. Петя и книга

Решение 
#include <iostream>

using namespace std;

int main() {
  int n; // число страниц в книге
  cin >> n;

  int A[7];
  for (int i = 0; i < 7; i++) {
    cin >> A[i];
  }

  int day = 0;

  while (n > 0) {
    for (int i = 0; n > 0 && i < 7; i++) {
      n -= A[i];
      if (n <= 0) {
        day = i + 1;
      }
    }
  }

  cout << day;
}
