Задача A. HQ9+

Решение 
#include <iostream>
#include <string>

using namespace std;

int main() {
  string s; // входная строка
  getline(cin, s);
  bool flag = false;

  for (char ch : s) {
    if (ch == 'H' || ch == 'Q' || ch == '9') {
      flag = true;
    }
  }

  if (flag) {
    cout << "YES";
  } else {
    cout << "NO";
  }
}
