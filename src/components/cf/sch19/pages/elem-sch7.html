Задача A. Камни на столе

Решение 
#include <iostream>

using namespace std;

int main() {
  // ввод данных
  int n; // число камней
  cin >> n;
  string s;
  cin >> s;

  // решение
  int res = 0;
  char c = s[0];
  for (int i = 1; i < s.size(); i++) {
    if (c == s[i]) {
      res++;
    }
    c = s[i];
  }

  // вывод результата
  cout << res;
}
