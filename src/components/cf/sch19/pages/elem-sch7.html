Задача A. Камни на столе

Решение 
#include <iostream>

using namespace std;

int main() {
  // ввод данных
  int n; // число камней
  cin >> n;
  string s;
  cin >> s;

  // решение
  int res = 0;
  char c = s[0];
  for (int i = 1; i < s.size(); i++) {
    if (c == s[i]) {
      res++;
    }
    c = s[i];
  }

  // вывод результата
  cout << res;
}

Задача B. Очередь в школе

Решение 
#include <iostream>

using namespace std;

int main() {
  int n; // число школьников от 1 до 50
  int t; // число секунд от 1 до 50
  cin >> n >> t;

  char s[51];
  for (int i = 1; i <= n; i++) {
    cin >> s[i];
  }

  for (int i = 1; i <= t; i++) {
    for (int j = 2; j <= n; j++) {
      if (s[j] == 'G' && s[j - 1] == 'B') {
        char tmp = s[j];
        s[j] = s[j - 1];
        s[j - 1] = tmp;
        j++;
      }
    }

  }

  for (int i = 1; i <= n; i++) {
    cout << s[i];
  }
}

Задача C. Несчастливый билет

Решение 
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  int n;
  cin >> n;
  string s;
  cin >> s;

  vector<int> A(n);
  vector<int> B(n);

  for (int i = 0; i < n; i++) {
    A[i] = s[i];
  }

  for (int i = 0; i < n; i++) {
    B[i] = s[i + n];
  }

  sort(A.begin(), A.end());
  sort(B.begin(), B.end());

  bool flag = true;

  if (A[n - 1] > B[n - 1]) {
    for (int i = 0; i < n; i++) {
      if (B[i] >= A[i]) {
        flag = false;
      }
    }
  } else if (B[n - 1] > A[n - 1]) {
    for (int i = 0; i < n; i++) {
      if (A[i] >= B[i]) {
        flag = false;
      }
    }
  } else {
    flag = false;
  }

  if (flag) {
    cout << "YES";
  } else {
    cout << "NO";
  }
}
