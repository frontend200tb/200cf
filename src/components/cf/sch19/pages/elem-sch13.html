Задача A. Красивый год

Решение 
#include <iostream>

using namespace std;

int main() {
  int y; // номер года от 1000 до 9000
  cin >> y;

  int res = y + 1;
  int d1 = res % 10;
  int d2 = (res / 10) % 10;
  int d3 = (res / 100) % 10;
  int d4 = res / 1000;

  while (d1 == d2 || d1 == d3 || d1 == d4
     || d2 == d3 || d2 == d4 || d3 == d4) {
    res++;
    d1 = res % 10;
    d2 = (res / 10) % 10;
    d3 = (res / 100) % 10;
    d4 = res / 1000;
  }

  cout << res;
}

Задача B. Выражение

Решение 
#include <iostream>
#include <algorithm>

using namespace std;

int main() {
  int a, b, c;
  cin >> a >> b >> c;

  int r1 = a + b + c;
  int r2 = a + b * c;
  int r3 = (a + b) * c;
  int r4 = a * b + c;
  int r5 = a * (b + c);
  int r6 = a * b * c;

  cout << max(r1, max(r2, max(r3, max(r4, max(r5, r6)))));
}

Задача D. Квадраты и кубы

Решение 
#include <iostream>
#include <set>

using namespace std;

int main() {
// условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n;
    cin >> n;

    set<int> st;
    // добавляем в сет квадраты
    for (int i = 1; i * i <= n; i++) {
      st.insert(i * i);
    }
    // добавляем в сет кубы
    for (int i = 1; i * i * i <= n; i++) {
      st.insert(i * i * i);
    }

    cout << st.size() << '\n';
  }
}
