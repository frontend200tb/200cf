Задача B. I Wanna Be the Guy

Решение
Задача 469A
Codeforces 469 Round 268 (Div. 2) 2014.09.20 
#include <iostream>

using namespace std;

int main() {
  // ввод данных
  int n; // число уровней
  int p; // уровни LittleX
  int q; // уровни LittleY
  cin >> n;

  // создадим частотный массив
  int F[101] = { 0 };
  int in;

  cin >> p;
  for (int i = 0; i < p; i++) {
    cin >> in;
    F[in]++;
  }

  cin >> q;
  for (int i = 0; i < q; i++) {
    cin >> in;
    F[in]++;
  }

  // решение
  bool flag = true; // они смогут пройти

  for (int i = 1; i <= n; i++) {
    if (F[i] == 0) {
      // если они не пройдут хотябы один уровень
      flag = false; // они не смогут пройти
    }
  }

  string yes = "I become the guy.";
  string no = "Oh, my keyboard!";

  // вывод результата
  if (flag) {
    cout << yes;
  } else {
    cout << no;
  }
}

Задача D. Шпион обнаружен!

Решение
Задача 1512A
Codeforces 1512 Round 713 (Div. 2) 2021.04.10 
#include <iostream>

using namespace std;

int main() {
  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // длина массива
    cin >> n;

    // всего встречается два числа
    int A[2] = { 0 }; // храним сами числа
    int B[2] = { 0 }; // сколько раз они встретились
    int C[2] = { 0 }; // позиция числа
    int in;

    for (int i = 1; i <= n; i++) {
      cin >> in;
      if (A[0] == 0) {
        A[0] = in;
        B[0] = 1;
        C[0] = i;
      } else if (A[0] == in) {
        B[0]++;
      } else if (A[1] == 0) {
        A[1] = in;
        B[1] = 1;
        C[1] = i;
      } else if (A[1] == in) {
        B[1]++;
      }
    }

    // вывод результата
    if (B[0] == 1) {
      cout << C[0] << '\n';
    } else {
      cout << C[1] << '\n';
    }
  }
}

Задача E. Командная олимпиада

Решение
Задача 490A
Codeforces 490 Round 279 (Div. 2) 2014.11.23 
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  // ввод данных
  int n; // число школьников
  cin >> n;
  vector<int> t1; // школьники со способностью к программированию
  vector<int> t2; // школьники со способностью к математике
  vector<int> t3; // школьники со способностью к физкультуре
  int in;

  for (int i = 1; i <= n; i++) {
    cin >> in;
    switch (in) {
    case 1: t1.push_back(i); break;
    case 2: t2.push_back(i); break;
    case 3: t3.push_back(i); break;
    }
  }

  // решение
  int size = min(t1.size(), min(t2.size(), t3.size()));

  // вывод результата
  if (size) {
    cout << size << '\n';
    for (int i = 0; i < size; i++) {
      cout << t1[i] << ' ' << t2[i] << ' ' << t3[i] << '\n';
    }
  } else {
    cout << 0;
  }
}

Решение
Задача 155А
Codeforces 155 Round 109 (Div. 2) 2012-02-24 
#include <iostream>

using namespace std;

int main() {
  // ввод данных
  int n; // число соревнований
  cin >> n;
  int in; // баллы за первое соревнование
  cin >> in;

  int min = in;
  int max = in;
  int cnt = 0;

  for (int i = 1; i < n; i++) {
    cin >> in;
    if (in > max) {
      cnt++;
      max = in;
    }
    if (in < min) {
      cnt++;
      min = in;
    }
  }

  // вывод результата
  cout << cnt;
}
