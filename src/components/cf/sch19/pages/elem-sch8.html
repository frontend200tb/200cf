Задача A. Линейная клавиатура

Решение 
#include <iostream>

using namespace std;

int main() {
  int t;
  cin >> t;

  while (t--) {
    string a, s;
    cin >> a >> s;

    int pos; // позиция буквы
    for (int i = 0; i < 26; i++) {
      if (a[i] == s[0]) {
        pos = i;
        break;
      }
    }
    int cnt = 0; // число букв между позициями
    for (int i = 1; i < s.size(); i++) {
      for (int j = 0; j < 26; j++) {
        if (a[j] == s[i]) {
          cnt += abs(j - pos);
          pos = j;
          break;
        }
      }
    }

    cout << cnt << '\n';
  }
}
