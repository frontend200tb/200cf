Задача A. Команда

Решение 
#include <iostream>

using namespace std;

int main() {
  int n; // число задач
  cin >> n;

  int cnt = 0; // число задач с решениями

  while (n--) {
    int a, b, c;
    cin >> a >> b >> c;
    if (a + b + c > 1) {
      cnt++;
    }
  }

  cout << cnt;
}
