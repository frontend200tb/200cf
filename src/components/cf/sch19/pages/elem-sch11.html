Решение 
#include <iostream>

using namespace std;

int main() {
  int n; // число сил
  cin >> n;
  int x, y, z; // вектор силы
  int sumx = 0, sumy = 0, sumz = 0;

  while (n--) {
    cin >> x >> y >> z;
    sumx += x;
    sumy += y;
    sumz += z;
  }

  if (sumx == 0 and sumy == 0 and sumz == 0) {
    cout << "YES";
  } else {
    cout << "NO";
  }
}
