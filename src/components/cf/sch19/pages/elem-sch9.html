#include <iostream>

using namespace std;

int main() {
  int n; // от 1 до 1000
  cin >> n;

  bool flag = false;
  if (n % 4 == 0 or n % 7 == 0 or n % 47 == 0 or n % 74 == 0 or
    n % 447 == 0 or n % 474 == 0 or n % 477 == 0 or
    n % 744 == 0 or n % 747 == 0 or n % 774 == 0) {
    flag = true;
  }

  if (flag) {
    cout << "YES";
  } else {
    cout << "NO";
  }
}
