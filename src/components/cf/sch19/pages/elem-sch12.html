Задача C. Квадратная строка?

Решение 
#include <iostream>

using namespace std;

int main() {
  int t;
  cin >> t;

  while (t--) {
    string s;
    cin >> s;

    bool flag = true;

    if (s.length() % 2 != 0) {
      flag = false;
    } else {
      for (int i = 0; flag and (i < s.length() / 2); i++) {
        if (s[i] != s[s.length() / 2 + i]) {
          flag = false;
        }
      }
    }

    if (flag) {
      cout << "YES\n";
    } else {
      cout << "NO\n";
    }
  }
}
