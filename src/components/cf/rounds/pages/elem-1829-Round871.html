Задача C. Мистер совершенство

Решение

Виктор может получить оба навыка только в двух случааях

  1. если прочитает две книги, одна из которых дает первый навык, но не дает второй, а другая дает второй навык, но не дает первый. Это книги с двоичным кодом '10' и '01'
  2. если прочитает одну книгу, которая дает и первый навык и второй навык. Это книга с кодом '11'

Можно не рассматривать книги, чтение которых не дает никаких навыков. Это книги с двоичным кодом '00'

Найдем все книги с кодами 01, 10, 11 и выберем из них те которые можно прочесть за минимальное время

#include <iostream>

using namespace std;

int main() {
  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n;
    cin >> n;
    int min10 = 10000000;
    int min01 = 10000000;
    int min11 = 10000000;
    for (int i = 0; i < n; i++) {
      int m;
      string s;
      cin >> m >> s;

      // решение
      if (s == "11") {
        min11 = min(min11, m);
      } else if (s == "10") {
        min10 = min(min10, m);
      } else if (s == "01") {
        min01 = min(min01, m);
      }
    }
    int result = min(min11, min10 + min01);

    // вывод результата
    if (result >= 1000000) {
      cout << -1 << '\n';
    } else {
      cout << result << '\n';
    }
  }
}

Задача D. Золотая лихорадка

Решение

Напишем рекурсивную функцию three(n, m). Если n == m то возвращаем true. Если m > n или n не делится на 3 то возвращаем false. Вызываем саму себя для (1/3)n и (2/3)n.

#include <iostream>

using namespace std;

bool three(int n, int m) {
  if (n == m) {
    return true;
  }
  int a, b;
  if (n % 3 || m > n) {
    return false;
  } else {
    a = n / 3;
    b = n / 3 * 2;
    return (three(a, m) || three(b, m));
  }
}

int main() {
  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n, m;
    cin >> n >> m;

    // решение
    bool gold = three(n, m);

    // вывод результата
    if (gold) {
      cout << "YES" << '\n';
    } else {
      cout << "NO" << '\n';
    }
  }
}

Задача G. Пирамида из банок

Решение 
#include <iostream>
#include <vector>

using namespace std;

const int max_n = 2000007;
const int max_i = 1500;

long long a[max_i][max_i] = {};
long long cur = 1;

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  vector<long long> DP(max_n);
  for (int i = 1; i < max_i; i++) {
    for (int j = i - 1; j >= 1; j--) {
      a[j][i - j] = a[j - 1][i - j] + a[j][i - j - 1]
      - a[j - 1][i - j - 1] + cur * cur;
      DP[cur] = a[j][i - j];
      cur++;
    }
  }

  int t; // число тестов
  cin >> t;

  while (t--) {
    int n; // номер банки
    cin >> n;

    cout << DP[n] << '\n';
  }
}
