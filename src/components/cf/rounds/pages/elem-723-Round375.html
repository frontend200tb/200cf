Задача A. Новый год: встреча друзей

Ограничение по времени 1с
Ограничение по памяти 256мб
Решение 
#include <iostream>
#include <algorithm>

using namespace std;

int main() {
  // ввод данных
  int x1, x2, x3; // три числа
  cin >> x1 >> x2 >> x3;

  // решение
  int res = max(x1, max(x2, x3)) - min(x1, min(x2, x3));

  // вывод результата
  cout << res << '\n';
}
