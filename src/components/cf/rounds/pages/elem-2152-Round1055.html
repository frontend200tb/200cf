Задача A. Увеличить или разбить

Решение

#include <iostream>
#include <vector>

using namespace std;

int main() {
  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // длина массива из нулей
    cin >> n;
    vector<int> A(n);
    for (int i = 0; i < n; i++) {
      cin >> A[i];
    }

    // решение
    int res = 0;
    // ответ будет равен числу различных чисел
    // плюс число обнулений, которых на 1 меньше чем различных чисел
    // соберем частотный массив
    vector<int> F(101);
    for (int i = 0; i < n; i++) {
      F[A[i]]++;
    }

    for (int i = 1; i < 101; i++) {
      if (F[i]) {
        res++;
      }
    }

    res = 2 * res - 1;

    // вывод результата
    cout << res << '\n';
  }
}

Задача B. Поймать Кругу

Решение

#include <iostream>
#include <algorithm>

using namespace std;

int main() {
  // число тестов
  int t;
  cin >> t;

  while (t--) {
    // ввод данных
    int n; // размер доски n+1 на n+1
    int rk, ck; // координаты Круги
    int rd, cd; // коориднаты Дорана
    cin >> n >> rk>> ck>> rd >> cd;

    // решение
    int res = 0; // число ходов Дорана
    int r; // расстояние по рядам
    int c; // расстояние по колонкам

    // Доран должен идти в сторону Круги
    // число его ходов равно числу клеток до края
    if (rd > rk) {
      r = rd;
    } else {
      r = n - rd;
    }

    if (cd > ck) {
      c = cd;
    } else {
      c = n - cd;
    }

    if (rd == rk) {
      res = c;
    } else if (cd == ck) {
      res = r;
    } else {
      res = max(r, c);
    }

    // вывод результата
    cout << res << '\n';
  }
}
