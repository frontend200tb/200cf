#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  int n; // число грамм теста
  int m; // число видов начинки
  int c0; // число грамм теста на булочку без начинки
  int d0; // цена булочки без начинки
  cin >> n >> m >> c0 >> d0;

  vector<int> A(m+1); // число грамм i-ой начинки
  vector<int> B(m+1); // число грамм i-ой начинки для одной булочки
  vector<int> C(m+1); // число грамм теста с i-ой начинкой для одной булочки
  vector<int> D(m+1); // цена булочки с i-ой начинкой
  for (int i = 1; i <= m; i++) {
    cin >> A[i] >> B[i] >> C[i] >> D[i];
  }

  // DP[i] максимальное количество денег,
  // которое мы получим если используем i грамм теста
  vector<int> DP(n+1,0);

  for (int i = 1; i <= m; i++) {
    int maxb = A[i] / B[i]; // максимум булочек из i-ой начинки
    for (int j = 1; j <= maxb; j++) {
      for (int k = 0; k <= n - C[i]; k++) {
        DP[k] = max(DP[k + C[i]] + D[i], DP[k]);
      }
    }
  }

  int res = 0;
  for (int i = 0; i <= n; i++) {
    res = max(res, DP[i] + i / c0 * d0);
  }

  cout << res;
}
