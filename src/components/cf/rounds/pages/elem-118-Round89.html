Задача A. Упражнение на строки

Решение 
#include <iostream>
#include <vector>

using namespace std;

int main() {
  vector<char> res;
  vector<char> vow{ 'a','A','o','O','y','Y','e','E','u','U','i','I' };

  string s;
  cin >> s;
  for (int i = 0; i < s.size(); i++) {
    if (find(vow.begin(), vow.end(), s[i]) == vow.end()) {
      res.push_back('.');
      res.push_back(tolower(s[i]));
    }
  }

  for (int i = 0; i < res.size(); i++) {
    cout << res[i];
  }
}
