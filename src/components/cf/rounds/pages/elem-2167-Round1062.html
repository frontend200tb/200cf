Задача A. Квадрат?

Решение 
#include <iostream>

using namespace std;

int main() {
  int t;
  cin >> t;

  while (t--) {
    int a, b, c, d; // длины палочек
    cin >> a >> b >> c >> d;

    if (a == b && a == c && a == d) {
      cout << "YES\n";
    } else {
      cout << "NO\n";
    }
  }
}

Задача B. Твоё имя

Решение 
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  int q;
  cin >> q;

  while (q--) {
    int n; // длины строки
    cin >> n;
    string s, t;
    cin >> s >> t;

    vector<char> A(n);
    vector<char> B(n);
    for (int i = 0; i < n; i++) {
      A[i] = s[i];
      B[i] = t[i];
    }

    sort(A.begin(), A.end());
    sort(B.begin(), B.end());
    bool flag = true;

    for (int i = 0; i < n; i++) {
      if (A[i] != B[i]) {
        flag = false;
      }
    }

    if (flag) {
      cout << "YES\n";
    } else {
      cout << "NO\n";
    }
  }
}
