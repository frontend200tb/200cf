using namespace std;

int main() {
 // условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif

  int t;
  cin >> t;

  while (t--) {
    int n;
    cin >> n;

    if (n % 2 == 1) {
      cout << 0 << '\n';
    } else {
      cout << n / 4 + 1 << '\n';
    }
  }
}
