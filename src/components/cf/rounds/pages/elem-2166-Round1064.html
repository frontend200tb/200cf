Задача A. Одинаковая разница

Решение 
#include <iostream>
#include <string>

using namespace std;

int main() {
// условная компиляция
#ifdef _DEBUG
  freopen("input.txt", "rt", stdin);
  freopen("output.txt", "wt", stdout);
#endif
  int t;
  cin >> t;

  while (t--) {
    int n; // длина строки
    cin >> n;
    string s; // строка
    cin >> s;

    int res = 0;
    for (int i = n - 1; i > 0; i--) {
      if (s[i] != s[i - 1]) {
        s[i - 1] = s[i];
        res++;
      }
    }
    cout << res << '\n';
  }
}
