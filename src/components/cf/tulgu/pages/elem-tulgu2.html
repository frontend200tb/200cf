Задача D. Артем в армии

Решение 
#include <iostream>

using namespace std;

int main() {
  int n; // число приказов
  int k; // танк Артема
  cin >> n >> k;
  int a, b; // номера танков от 1 до 3

  for (int i = 0; i < n; i++) {
    cin >> a >> b;
    if (k == a) {
      k = b;
    } else if (k == b) {
      k = a;
    }
  }

  cout << k;
}
