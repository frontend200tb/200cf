Задача A. Мойка автомобиля

Решение 
#include <iostream>
#include <cmath>

using namespace std;

int main() {
  int x; // фиксированная оплата
  int y; // стоимость минуты
  int z; // стоимость литра
  int t; // число минут
  int v; // число литров
  cin >> x >> y >> z >> t >> v;

  int cost_min = y * t; // оплата за минуты
  int cost_lit = z * v; // оплата за литры
  int cost = min(x, min(cost_min, cost_lit));

  cout << cost;
}

Задача B. Настоящие друзья

Решение

Задача на бинарный поиск по ответу. Нужно найти такую коробку, чтобы число всех конфет сверху до этой коробки было равно числу конфет в нижних коробках.

Сумма чисел от 1 до n

n * (n+1) / 2

Сумма чисел от A до B

(B-A+1) * (A+B) / 2 
#include <iostream>

using namespace std;

typedef long long ll;

ll sum1(ll n) {
  return n * (n + 1) / 2;
}

ll sum2(ll A, ll B) {
  return (B - A + 1) * (A + B) / 2;
}

int main() {
  int t; // число тестов
  cin >> t;

  while (t--) {
    ll k; // число коробок
    cin >> k;

    // сумма чисел от 1 до n
    // n * (n+1) / 2
    // сумма чисел от A до B
    // (B-A+1) * (A+B) / 2

    ll left = 0;
    ll right = k + 1;
    ll mid;

    // бинарный поиск по ответу
    while (right - left > 1) {
      mid = (right + left) / 2;
      if (sum1(mid) <= sum2(mid+1, k)) {
        left = mid;
      } else {
        right = mid;
      }
    }

    if (sum1(left) == sum2(left+1, k)) {
      cout << "yes" << '\n';
    } else {
      cout << "no" << '\n';
    }
  }
}

Задача C. Пирамида

Решение

Создадим вектор кирпичей, для каждого кирпича будем хранить только его меньшую сторону. Отсортируем вектор кирпичей по размеру. Пройдем по всем кирпичам и если очередной кирпич может сделать еще один слой пирамиды, то будем увеличивать высоту пирамиды на один.

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  int n; // число кирпичей
  cin >> n;
  int a; // меньшая сторона кирпича
  int b; // большая стороная кирпича
  vector<int> A(n);
  for (int i = 0; i < n; i++) {
    cin >> a >> b; // стороны кирпича
    if (a > b) {
      A[i] = b; // размер кирпича
    } else {
      A[i] = a; // размер кирпича
    }
  }

  sort(A.begin(), A.end());
  int k = 0; // высота пирамиды

  for (int i = 0; i < n; i++) {
    if (A[i] >= 2 * (k + 1) - 1) {
      k++;
    }
  }

  cout << k;
}

Задача D. Простой феникс

Решение на 39 баллов из 100 потому что превышено ограниение по памяти 
#include <iostream>
#include <vector>

using namespace std;

int main() {
  int n, m; // размер поля
  int k; // число врагов
  cin >> n >> m >> k;
  int x; // координата x
  int y; // координата y

  vector<vector<int> > A(n+4, vector<int>(m+4));
  // 0 - свободно, 1 - враг
  // вокруг поля добавим по две свободные клетки

  for (int i = 0; i < k; i++) {
    cin >> x >> y; // координаты врага
    A[x+1][y+1] = 1;
  }

  bool res = 0; // нет эффективной атаки

  for (int i = 2; i < n + 2; i++) {
    for (int j = 2; j < m + 2; j++) {
      if (A[i][j] == 0) {
        if (A[i - 1][j] == 1 && A[i - 2][j] == 1) {
          res = 1;
        }
        if (A[i + 1][j] == 1 && A[i + 2][j] == 1) {
          res = 1;
        }
        if (A[i][j - 1] == 1 && A[i][j - 2] == 1) {
          res = 1;
        }
        if (A[i][j + 1] == 1 && A[i][j + 2] == 1) {
          res = 1;
        }
      }
    }
  }

  if (res) {
    cout << "yes";
  } else {
    cout << "no";
  }
}
