#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  freopen("ladder.in", "rt", stdin);
  freopen("ladder.out", "wt", stdout);

  int n; // число ступенек
  cin >> n;

  vector<int> A(n + 1);
  for (int i = 1; i <= n; i++) {
    cin >> A[i];
  }

  vector<int> DP(n + 1);
  DP[0] = 0;
  DP[1] = A[1];
  for (int i = 2; i <= n; i++) {
    DP[i] = max(DP[i - 2] + A[i], DP[i - 1] + A[i]);
  }

  cout << DP[n];
}

#include <iostream>
#include <vector>

using namespace std;

int main() {
  freopen("lepus.in", "rt", stdin);
  freopen("lepus.out", "wt", stdout);

  int n; // число клеток
  cin >> n;

  vector<char> A(n + 1);
  for (int i = 1; i <= n; i++) {
    cin >> A[i];
  }

  vector<int> DP(n + 1);
  DP[1] = 0;
  for (int i = 1; i < n; i++) {
    // проверяем что не осталось ходов
    if ( (n > i + 5 && A[i + 1] == 'w' && A[i + 3] == 'w' && A[i + 5] == 'w') ||
       (i + 4 == n && A[i + 1] == 'w' && A[i + 3] == 'w') ||
       (i + 2 == n && A[i + 1] == 'w') ) {
      cout << -1;
      return 0;
    }

    // если до конца остался один ход
    if (i + 1 == n) {
      // делаем один ход
      DP[i + 1] = DP[i];

      // если до конца осталось два хода
    } else if (i + 2 == n) {
      if (A[i + 1] == '"') {
        // делаем один ход
        DP[i + 1] = DP[i] + 1;
      } else if (A[i + 1] == '.') {
        // делаем один ход
        DP[i + 1] = DP[i];
      }

      // если до конца осталось три хода
    } else if (i + 3 == n) {
      if (A[i + 1] == 'w' || A[i+2] == 'w') {
        // делаем три хода
        DP[i + 3] = DP[i];
        i = i + 2;
      } else if (A[i + 1] == '"') {
        // делаем один ход
        DP[i + 1] = DP[i] + 1;
      } else if (A[i + 1] == '.') {
        // делаем один ход
        DP[i + 1] = DP[i];
      }

      // если до конца осталось четыре хода
    } else if (i + 4 == n) {
      if (A[i + 1] == 'w') {
        // делаем три хода
        if (A[i + 3] == '"') {
          DP[i + 3] = DP[i] + 1;
          i = i + 2;
        } else if (A[i + 3] == '.') {
          DP[i + 3] = DP[i];
          i = i + 2;
        }
      } else if (A[i + 1] == '"') {
        // делаем один ход
        DP[i + 1] = DP[i] + 1;
      } else if (A[i + 1] == '.') {
        // делаем один ход
        DP[i + 1] = DP[i];
      }

      // если до конца осталось пять ходов
    } else if (i + 5 == n) {
      if ( (A[i + 2] == 'w' && A[i + 4] == 'w') ||
         ( A[i + 1] == 'w' && A[i + 3] == 'w') ||
         ( A[i + 1] == 'w' && A[i + 4] == 'w') ) {
        // делаем пять ходов
        DP[i + 5] = DP[i];
        i = i + 4;
      } else if (A[i + 1] == 'w') {
        // делаем три хода
        if (A[i + 3] == '"') {
          DP[i + 3] = DP[i] + 1;
          i = i + 2;
        } else if (A[i + 3] == '.') {
          DP[i + 3] = DP[i];
          i = i + 2;

      } else if (A[i + 1] == '"') {
        // делаем один ход

      } else if (A[i + 1] == '.') {
        // делаем один ход
        DP[i + 1] = DP[i];

      // проверяем что можно съесть траву на ходе +1

    } else if ( (A[i + 1] == '"') &&

    } else if ( (A[i + 1] == '.' && i + 2 <= n && A[i + 2] == '"') &&

    } else if (i + 8 <= n && A[i + 8] == '"' && A[i + 3] == '.' &&
      A[i + 1] != '"' && A[i + 2] != '"' && A[i + 4] != '"' &&
      A[i + 5] != '"' && A[i + 6] != '"' && A[i + 7] != '"') {
      // делаем три хода
      DP[i + 3] = DP[i];

    } else if ( (A[i + 1] == '.') &&

    } else if ( (i + 3 <= n && A[i + 3] == '.') &&
