#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
  freopen("ladder.in", "rt", stdin);
  freopen("ladder.out", "wt", stdout);

  int n; // число ступенек
  cin >> n;

  vector<int> A(n + 1);
  for (int i = 1; i <= n; i++) {
    cin >> A[i];
  }

  vector<int> DP(n + 1);
  DP[0] = 0;
  DP[1] = A[1];
  for (int i = 2; i <= n; i++) {
    DP[i] = max(DP[i - 2] + A[i], DP[i - 1] + A[i]);
  }

  cout << DP[n];
}

int main() {
  freopen("lepus.in", "rt", stdin);
  freopen("lepus.out", "wt", stdout);

  int n; // число клеток
  cin >> n;

  vector<char> A(n + 1);
  for (int i = 1; i <= n; i++) {
    cin >> A[i];
  }

  vector<int> DP(n + 1);
  DP[1] = 0;
  for (int i = 1; i < n; i++) {
    if (A[i + 1] == '"') {
      DP[i+1] = DP[i] + 1;
    } else if (i + 3 <= n && A[i + 3] == '"') {
      DP[i] = DP[i] + 1;
      i = i + 2;
    } else if (i + 5 <= n && A[i + 5] == '"') {
      DP[i] = DP[i] + 1;
      i = i + 4;
    } else if (A[i + 1] == '.') {
      DP[i+1] = DP[i];
    } else if (i + 3 <= n && A[i + 3] == '.') {
      DP[i+3] = DP[i];
      i = i + 2;
    } else if (i + 5 <= n && A[i + 5] == '.') {
      DP[i+5] = DP[i];
      i = i + 4;
    } else {
      cout << -1;
      return 0;
    }
  }

  cout << DP[n];
}
